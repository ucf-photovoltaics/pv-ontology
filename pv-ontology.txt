mds:Cell a owl:Class ;
    rdfs:label "Cell" ;
    owl:disjointWith mds:PhotovoltaicModule ;
    rdfs:subClassOf cco:ont00000995 ,
                    mds:Sample ;
    skos:altLabel "Cell" ;
    skos:definition "A photovoltaic (PV) cell is a semiconductor device that converts optical power (i.e., $P_{in} = P_{opt}$) into electrical power (i.e., $P_{out} = P_{elect}$). A semiconductor absorber with the appropriate bandgap ($E_{g}$), selected to maximize power conversion from the solar spectrum, converts incident photons into electronic charge carriers (i.e., negatively charged electron and positively charged holes, or missing electrons in the valence band). Other parts (i.e., subcomponents or device building blocks) help transfer the charge carriers and eventually the current ($I$) and voltage ($V$) to an external circuit where energy ($E_{out}$) can be performed with this electrical power ($P_{out} = I cdot V$) over time, i.e., $E_{out} = int P_{out}(t) cdot dt$. These subcomponents help minize power conversion losses and therefore maximize the photoconversion efficiency ($eta = frac{P_{out}}{P_{in}}$). These PV cells are normally encapsulated and then electrically connected together in series (typically) or series-parallel configurations to to form PV module" ;
    mds:hasDomain "BuiltEnv" ;
    mds:hasSubDomain "PV-Module" .
